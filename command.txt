# display all
curl -i -H "Content-Type: application/json" -X GET http://localhost:5000/todos/

# detail task
curl -i -H "Content-Type: application/json" -X GET http://localhost:5000/todos/1

# Create
curl -i -H "Content-Type: application/json" -X POST -d '{"title":"Bread", "content": "Hello, world!"}' http://localhost:5000/todos/

# Update
curl -i -H "Content-Type: application/json" -X PUT -d '{"title":"Bread", "content": "Hello, world!", "done": "False"}' http://localhost:5000/todos/2

# delete
curl -i -H "Content-Type: application/json" -X DELETE http://localhost:5000/todos/20
